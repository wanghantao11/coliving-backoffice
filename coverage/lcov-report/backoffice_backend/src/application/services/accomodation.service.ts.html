import { inject, injectable } from 'inversify'
import 'reflect-metadata'
import { get } from 'lodash'
 
import { AccomodationDao } from './../../domain/dao/accomodation.dao'
import { ApplicationDao } from './../../domain/dao/application.dao'
import { Accomodation } from './../../domain/entity/accomodation'
import { parseImagePath } from './../../infrastructure/utils/firebase'
import { isPublishable } from './../middleware/validator/accommodation'
import { deleteImage } from './../firebase/accomodation'
 
@injectable()
export class AccomodationService {
  constructor(
    @inject('AccomodationDao') private accomodationDao: AccomodationDao,
    @inject('ApplicationDao') private applicationDao: ApplicationDao
  ) {}
 
  public createAccomodation = (acc: Accomodation) => this.accomodationDao.createAccomodation(acc)
 
  public updateAccomodation = (acc: Accomodation) => this.accomodationDao.updateAccomodation(acc.id, acc)
 
  public getPublishedAccomodation = (id: string) =>
    Promise.all([
      this.accomodationDao.getPublishedAccomodation(Number(id)),
      this.applicationDao.getCountOfApplications([Number(id)]),
    ]).then(([acc, apps]) => ({
      ...acc, applicationsMade: get(apps.find(app => app.accommodationId === Number(id)), 'applicationsMade', 0),
    }))
 
  public getAccomodation = (id: string) => this.accomodationDao.getAccomodation(Number(id))
 
  public getMyAccomodations = (clientId: number, isPublished: boolean, offset: string, limit: number = 12) =>
    this.accomodationDao.getMyAccomodations(clientId, isPublished, limit, Number(offset))
      .then(projects => projects.length === 0 ? Promise.reject({ message: 'NOT_FOUND' }) : projects)
      .then(projects => this.applicationDao.getCountOfApplications(projects.map(project => project.id))
        .then(apps =>
          projects.map(project => ({
            ...project,
            applicationsMade: get(apps.find(app => app.accommodationId === project.id), 'applicationsMade', 0),
          }))))
      .catch(() => ([]))
 
  public getAllMyAccomodations = (clientId: number, offset: string, limit: number = 12) =>
    this.accomodationDao.getAllMyAccomodations(clientId, limit, Number(offset))
      .then(projects => projects.length === 0 ? Promise.reject({ message: 'NOT_FOUND' }) : projects)
      .then(projects => this.applicationDao.getCountOfApplications(projects.map(project => project.id))
        .then(apps =>
          projects.map(project => ({
            ...project,
            applicationsMade: get(apps.find(app => app.accommodationId === project.id), 'applicationsMade', 0),
          }))))
 
  public getAllPublishedAccomodations = (offset: string, limit: number = 12) =>
    this.accomodationDao.getAllPublishedAccomodations(limit, Number(offset))
      .then(projects => projects.length === 0 ? Promise.reject({ message: 'NOT_FOUND' }) : projects)
      .then(projects => this.applicationDao.getCountOfApplications(projects.map(project => project.id))
        .then(apps => projects.map(project => ({
          ...project,
          applicationsMade: get(apps.find(app => app.accommodationId === project.id), 'applicationsMade', 0),
        }))))
 
  public publishAccomodation = (id: string): Promise<boolean> =>
    this.accomodationDao.getAccomodation(Number(id))
      .then(project => !project ? Promise.reject({ message: 'NOT_FOUND' }) : project)
      .then(project => project.published ? Promise.reject({ message: 'CONFLICT' }) : project)
      .then(project => isPublishable(project)
        .then(() => this.accomodationDao.updateAccomodation(project.id, {
          published_at: new Date(),
          published: true,
        })))
      .then(() => true)
 
  public deleteAccomodation = (id: string): Promise<number> =>
    this.accomodationDao.deleteAccomodation(Number(id)).then(() => id)
}
